database model with mongodb for the connect app

USERS
{
    _id: ObjectId,
    email: string,
    password_hash: string,
    first_name: string,
    last_name: string,
    birth_date: date,
    profile_picture: string,
    bio: string,
    roles: array[string],
    tutor_profile: {
        description: string,
        subjects: array[ObjectId],
        certifications: array[string],
        average_rating: float,
        reviews_count: int,
        cv_path: string,
    } | null
    payment_methods: array[{
        token_id: string,
        type: string,
        expiration_date: string // Mese/Anno (es: "12/28"),
        provider: string,
        last_four_digit: string,
        is_default: boolean
    }]
    coins: int,
    creation_date: date,
    last_access: date
}

SUBJECTS
{

    field: string
}

CLASSROOMS
{
    _id: ObjectId,
    name: string,
    members: [
    { user_id: ObjectId, role: string, join_date: date }
    ],

    invite_code: string,
    creation_date: date
}

CLASSROOMS_DOCUMENTS
{
    _id: ObjectId,
    classroom_id: ObjectId,

    upload_date: date
}

CONVERSATIONS
{
    _id: ObjectId,
    type: string,
    classroom_id: ObjectId | null,
    participants: array[ObjectId],
    creation_date: date,
    last_activity: date
}

MESSAGES
{
    _id: ObjectId,
    conversation_id: ObjectId,
    sender_id: ObjectId,
    text: string,
    attachments: array[string],
    creation_date: date
}

TESTS
{
    _id: ObjectId,
    title: string,
    classroom_id: ObjectId,
    creator_id: ObjectId,
    questions: [{
        _id: ObjectId,

        question: string,
        answers: array[string],
        correct_answer: int | null
    }],
    time_limit: int
}

TESTS_SUBMISSIONS
{

REVIEWS

COINS_TRANSACTIONS
